% pdf-header.tex —— build_pdf.sh 專用的 xelatex 補字設定
% 目的:Noto Sans CJK TC 缺少部分「技術符號」(統計下標、勾叉),
%       用 newunicodechar 針對這些字元 fallback 到 DejaVu(覆蓋完整)。
% CJK 本體與斷行仍由 pandoc 的 xeCJK(CJKmainfont)負責,這裡只補符號。

\usepackage{newunicodechar}

% 補字用的 fallback 字型(DejaVu 對數學/技術符號覆蓋完整),縮放對齊小寫高度
\newfontfamily\symfallback{DejaVu Sans}[Scale=MatchLowercase]

% 下標數字 ₀–₉(U+2080–2089)
\newunicodechar{₀}{{\symfallback ₀}}
\newunicodechar{₁}{{\symfallback ₁}}
\newunicodechar{₂}{{\symfallback ₂}}
\newunicodechar{₃}{{\symfallback ₃}}
\newunicodechar{₄}{{\symfallback ₄}}
\newunicodechar{₅}{{\symfallback ₅}}
\newunicodechar{₆}{{\symfallback ₆}}
\newunicodechar{₇}{{\symfallback ₇}}
\newunicodechar{₈}{{\symfallback ₈}}
\newunicodechar{₉}{{\symfallback ₉}}
\newunicodechar{₋}{{\symfallback ₋}}   % 下標減號 U+208B
\newunicodechar{ᵢ}{{\symfallback ᵢ}}   % 下標 i U+1D62(統計 Σxᵢ)

% 勾 / 叉(表格常用)
\newunicodechar{✓}{{\symfallback ✓}}   % U+2713
\newunicodechar{✔}{{\symfallback ✔}}   % U+2714
\newunicodechar{✗}{{\symfallback ✗}}   % U+2717
\newunicodechar{✘}{{\symfallback ✘}}   % U+2718
